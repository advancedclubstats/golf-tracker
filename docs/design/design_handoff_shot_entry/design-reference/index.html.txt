<script type="text/babel" data-presets="react">
  const TWEAK_DEFAULTS = /*EDITMODE-BEGIN*/{
    "density": "comfy",
    "radius": 18,
    "lime": true,
    "night": false
  }/*EDITMODE-END*/;

  function App() {
    const [t, setTweak] = useTweaks(TWEAK_DEFAULTS);

    React.useEffect(() => {
      document.body.classList.toggle("night", !!t.night);
    }, [t.night]);

    // scale the fixed 402×874 device to fit the viewport
    React.useEffect(() => {
      const fit = () => {
        const s = Math.min(1, (window.innerHeight - 32) / 874, (window.innerWidth - 32) / 402);
        document.documentElement.style.setProperty("--scale", s.toFixed(3));
      };
      fit(); window.addEventListener("resize", fit);
      return () => window.removeEventListener("resize", fit);
    }, []);

    return (
      <React.Fragment>
        <div className="stage">
          <IOSDevice dark={t.night}>
            <ShotFlow t={t} />
          </IOSDevice>
        </div>

        <TweaksPanel>
          <TweakSection label="Feel" />
          <TweakRadio label="Density" value={t.density} options={["compact", "regular", "comfy"]}
            onChange={(v) => setTweak("density", v)} />
          <TweakSlider label="Corner radius" value={t.radius} min={8} max={28} step={1} unit="px"
            onChange={(v) => setTweak("radius", v)} />
          <TweakSection label="Brand" />
          <TweakToggle label="Lime accent" value={t.lime}
            onChange={(v) => setTweak("lime", v)} />
          <TweakToggle label="Night (clubhouse)" value={t.night}
            onChange={(v) => setTweak("night", v)} />
        </TweaksPanel>
      </React.Fragment>
    );
  }

  ReactDOM.createRoot(document.getElementById("root")).render(<App />);
</script>
